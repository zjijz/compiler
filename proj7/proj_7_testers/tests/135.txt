# Tests static variables that turn into dynamic through a block (similar to the read test that did this before)
# Input: 4
# Output: 510.0trueTrue

begin
    int x := 10;
    float f := 20.0;
    string s := "hey";
    bool b := True;
    write(x, f, s, b);

    int a;
    read(a);
    if not (a < 1) then begin
        x := 5;
        f := 10.0;
        s := "true";
        b := True;
    end else begin
        x := 20;
        f := 40.0;
        s := "false";
        b := False;
    end

    write(x, f, s, b);
end
